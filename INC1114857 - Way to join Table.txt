SELECT DISTINCT
    wip_entity_name,
    oola.line_number
FROM
         apps.oe_order_headers_all ooha
    INNER JOIN apps.oe_order_lines_all oola ON oola.header_id = ooha.header_id
    LEFT JOIN apps.wip_entities       wip ON wip.organization_id = oola.ship_from_org_id
                                       AND wip.primary_item_id = oola.inventory_item_id
WHERE
    
    ooha.order_number = 1050317026;

select
we.wip_entity_name,
oola.line_number
From
oe_order_headers_all ooha,
oe_order_lines_all oola,
mtl_reservations mr,
wip_discrete_jobs wdj,
wip_entities we
Where
1=1
AND ooha.header_id = oola.header_id
AND mr.demand_source_line_id = oola.line_id
AND mr.supply_source_header_id = we.wip_entity_id
AND we.wip_entity_id = wdj.wip_entity_id
AND oola.ship_from_org_id = we.organization_id
AND we.primary_item_id = oola.inventory_item_id
AND we.organization_id = wdj.organization_id
AND ooha.order_number = '1050317026';
